server {
    listen 443 ssl;
    server_name khaothi.fermat.vn;

    ssl_certificate /etc/letsencrypt/live/khaothi.fermat.vn/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/khaothi.fermat.vn/privkey.pem;

    client_max_body_size 100M;

    root /opt/khaothi.fermat.vn/frontend/dist;
    index index.html;

    location /static/ {
        alias /opt/khaothi.fermat.vn/backend/staticfiles/;
    }

    location /api/ {
        proxy_pass http://127.0.0.1:8000/api/;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto https;
        proxy_set_header X-Forwarded-Host $host;
        proxy_set_header X-Forwarded-Port 443;
        proxy_redirect off;
    }

    location /django-admin/ {
        proxy_pass http://127.0.0.1:8000/django-admin/;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto https;
        proxy_set_header X-Forwarded-Host $host;
        proxy_set_header X-Forwarded-Port 443;
        proxy_redirect off;
    }

    location / {
        try_files $uri $uri/ /index.html;
    }
}
server {
    server_name workspace.fermat.vn;

    location / {
        proxy_pass http://127.0.0.1:5500; # Trỏ về cổng ứng dụng của bạn
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/workspace.fermat.vn/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/workspace.fermat.vn/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot




}
